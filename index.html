Count Characters

    const countCharacters = (S) => 
    {
        let countA=0;
        let countD=0;
        let B=[];
        for(let i=0;i < S.length;i++){
            if(S.charAt(i)==="A"){
              countA++;
            }
            else if(S.charAt(i)==="D"){
              countD++;
            }
        }
        B[0]=countA;
        B[1]=countD;
        return B;
    };

        }
        }

        res+=str[i]+c;

    }

    };
